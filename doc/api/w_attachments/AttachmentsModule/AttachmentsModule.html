AttachmentsModule(
    {@required ExtensionContext extensionContext,
    @required Session session,
    @required String this.documentId,
    AttachmentsLoaderFactory attachmentLoaderFactory,
    CustomDisplayStrings customStrings,
    AttachmentsService attachmentsService,
    List<Bundle> initialAttachments,
    List<ContextGroup> initialGroups,
    List<Filter> initialFilters,
    int apiLoadAttachmentsDebounceMillis: 0,
    bool disableSelection: false,
    bool this.disableRenames: false,
    int this.numRenderersFromBottomToTriggerLoading: 0}) {
  moduleActions = new ModuleActions();
  attachmentActions = new AttachmentActions();

  attachmentsService =
      attachmentsService ?? new AttachmentsService(bigskyClient: session.createBigskyClient());

  _events = new AttachmentsEvents();
  _store = new AttachmentsStore(
      documentId: this.documentId,
      moduleActions: moduleActions,
      attachmentActions: attachmentActions,
      attachmentsEvents: _events,
      dispatchKey: attachmentsModuleDispatchKey,
      attachmentsService: attachmentsService,
      extensionContext: extensionContext,
      attachments: initialAttachments ?? [],
      groups: initialGroups ?? [],
      initialFilters: initialFilters ?? []);
  _components = new AttachmentsComponents(
      store: _store, moduleActions: moduleActions, attachmentActions: attachmentActions);
  _api = new AttachmentsApi(attachmentActions, moduleActions, _store, attachmentLoaderFactory,
      apiLoadAttachmentsDebounceMillis);
}
